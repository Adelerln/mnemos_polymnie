-- WARNING: This schema is for context only and is not meant to be run.
-- Table order and constraints may not be valid for execution.

CREATE TABLE public.adults (
  id bigint NOT NULL DEFAULT nextval('adults_id_seq'::regclass),
  civility text,
  last_name text NOT NULL,
  first_name text NOT NULL,
  address text,
  complement text,
  postal_code text,
  city text,
  country text,
  phone_1 text,
  phone_2 text,
  email text,
  partner_id bigint,
  vacaf_number text,
  notes text,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  anonymized_at timestamp with time zone,
  CONSTRAINT adults_pkey PRIMARY KEY (id),
  CONSTRAINT adults_partner_id_fkey FOREIGN KEY (partner_id) REFERENCES public.partners(id)
);
CREATE TABLE public.allowed_emails (
  email text NOT NULL,
  CONSTRAINT allowed_emails_pkey PRIMARY KEY (email)
);
CREATE TABLE public.centres (
  id bigint NOT NULL DEFAULT nextval('centres_id_seq'::regclass),
  name text NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  created_by uuid,
  modified_at timestamp with time zone,
  modified_by uuid,
  is_active boolean NOT NULL DEFAULT true,
  address_street text,
  address_extra text,
  postal_code text,
  city text,
  phone_landline text,
  generic_email text,
  ddcs_number text,
  gps_latitude numeric,
  gps_longitude numeric,
  commission_pv_date date,
  commission_expiry_date date,
  CONSTRAINT centres_pkey PRIMARY KEY (id)
);
CREATE TABLE public.children (
  id text NOT NULL,
  last_name text,
  first_name text,
  birth_date text,
  gender text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  updated_at timestamp with time zone NOT NULL DEFAULT now(),
  allergies text,
  diet text,
  health_issues text,
  instructions text,
  transport_notes text,
  friend text,
  family_id bigint NOT NULL,
  anonymized_at timestamp with time zone,
  CONSTRAINT children_pkey PRIMARY KEY (id),
  CONSTRAINT children_family_id_fkey FOREIGN KEY (family_id) REFERENCES public.families(id)
);
CREATE TABLE public.contacts_centres (
  id bigint NOT NULL DEFAULT nextval('contacts_centres_id_seq'::regclass),
  centre_id bigint NOT NULL,
  civility text,
  last_name text NOT NULL,
  first_name text NOT NULL,
  role text,
  phone_1 text,
  phone_2 text,
  email text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT contacts_centres_pkey PRIMARY KEY (id),
  CONSTRAINT contacts_centres_centre_id_fkey FOREIGN KEY (centre_id) REFERENCES public.centres(id)
);
CREATE TABLE public.families (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  id_client text NOT NULL UNIQUE,
  email text,
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  updated_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  user_id uuid,
  label text,
  notes text,
  anonymized_at timestamp with time zone,
  CONSTRAINT families_pkey PRIMARY KEY (id)
);
CREATE TABLE public.family_adults (
  id bigint NOT NULL DEFAULT nextval('family_adults_id_seq'::regclass),
  family_id bigint NOT NULL,
  adult_id bigint NOT NULL,
  role text NOT NULL,
  is_primary boolean DEFAULT false,
  position smallint,
  can_be_contacted boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT now(),
  can_be_contacted_global boolean DEFAULT false,
  CONSTRAINT family_adults_pkey PRIMARY KEY (id),
  CONSTRAINT family_adults_family_id_fkey FOREIGN KEY (family_id) REFERENCES public.families(id),
  CONSTRAINT family_adults_adult_id_fkey FOREIGN KEY (adult_id) REFERENCES public.adults(id)
);
CREATE TABLE public.identifications (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  user_id uuid,
  email text NOT NULL,
  status text NOT NULL CHECK (status = ANY (ARRAY['success'::text, 'failure'::text, 'locked'::text])),
  ip_address inet,
  user_agent text,
  context jsonb DEFAULT '{}'::jsonb,
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  CONSTRAINT identifications_pkey PRIMARY KEY (id),
  CONSTRAINT identifications_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)
);
CREATE TABLE public.partners (
  id bigint NOT NULL DEFAULT nextval('partners_id_seq'::regclass),
  name text NOT NULL,
  type USER-DEFINED DEFAULT 'AUTRE'::partner_type,
  address text,
  postal_code text,
  city text,
  country text,
  email text,
  phone text,
  notes text,
  created_at timestamp with time zone DEFAULT now(),
  updated_at timestamp with time zone DEFAULT now(),
  CONSTRAINT partners_pkey PRIMARY KEY (id)
);
